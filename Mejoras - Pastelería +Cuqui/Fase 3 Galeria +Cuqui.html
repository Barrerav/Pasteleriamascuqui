Contadores animados — fix

El problema actual: los contadores arrancan en 0 aunque ya sean visibles, y si el IntersectionObserver no dispara (por scroll rápido o móvil), se quedan en 0 para siempre. Esta versión lo arregla.

Fix Dos problemas, una solución ~30 min

Problema 1: Los contadores se activan aunque el usuario ya los vea al cargar la página (sin hacer scroll).
Problema 2: Si el usuario scrollea muy rápido y el observer no dispara en tiempo, los números se quedan en 0.
Solución: IntersectionObserver con threshold: 0.2 + fallback automático de 2 segundos.

Regla: Si el elemento ya es visible al cargar → animar inmediatamente. Si entra por scroll → animar al entrar. Si pasan 2s sin animarse → mostrar valor final directo.

src/components/Gallery.tsx — galería con filtros
'use client'
import { useState } from 'react'
import Image from 'next/image'
import Lightbox from 'yet-another-react-lightbox'
import 'yet-another-react-lightbox/styles.css'
import { galleryPhotos } from '@/data/photos'
import './gallery.css'

// Categorías para el filtro
const CATS = [
  { id: 'all',       label: 'Todos'     },
  { id: 'cafeteria', label: '☕ Cafetería'  },
  { id: 'pasteleria',label: '🎂 Pastelería' },
  { id: 'desayunos', label: '🥐 Desayunos'  },
  { id: 'eventos',   label: '🎉 Eventos'    },
]

export default function Gallery() {
  const [cat, setCat]   = useState('all')
  const [idx, setIdx]   = useState(-1)  // -1 = lightbox cerrado

  const filtered = galleryPhotos
    .filter(p => cat === 'all' || p.cat === cat)

  return (
    <section className="gallery-section" id="galeria">
      <p className="section-eyebrow">Nuestros momentos</p>
      <h2 className="section-title">Galería</h2>

      {"/* ── Filtros ── */"}
      <div className="filter-tabs">
        {CATS.map(c => (
          <button
            key={c.id}
            className={`filter-btn ${cat === c.id ? 'active' : ''}`}
            onClick={() => setCat(c.id)}
          >{c.label}</button>
        ))}
      </div>

      {"/* ── Grid de fotos ── */"}
      <div className="gallery-grid">
        {filtered.map((photo, i) => (
          <div
            key={photo.src}
            className="g-cell"
            onClick={() => setIdx(i)}  // ← abre el lightbox
          >
            <Image
              src={photo.src} alt={photo.alt}
              fill className="g-img"
              sizes="(max-width:768px) 50vw, 25vw"
            />
          </div>
        ))}
      </div>

      {"/* ── Lightbox ── */"}
      <Lightbox
        open={idx >= 0}
        index={idx}
        close={() => setIdx(-1)}
        slides={filtered.map(p => ({ src: p.src, alt: p.alt }))}
      />

    </section>
  )
}

src/hooks/useCounter.ts — hook reutilizable con fallback
'use client'
import { useEffect, useRef, useState } from 'react'

interface Options {
  target:    number   // valor final
  duration?: number   // ms de animación (default 1800)
  decimals?: number   // decimales (default 0)
  fallback?: number   // ms antes de mostrar valor final (default 2000)
}

export function useCounter(ref: React.RefObject<Element>, opts: Options) {
  const { target, duration = 1800, decimals = 0, fallback = 2000 } = opts
  const [value, setValue] = useState(0)
  const animated = useRef(false)

  const animate = () => {
    if (animated.current) return
    animated.current = true
    const start = performance.now()
    const tick = (now: number) => {
      const progress = Math.min((now - start) / duration, 1)
      const ease = 1 - Math.pow(1 - progress, 3)  // easeOutCubic
      setValue(parseFloat((ease * target).toFixed(decimals)))
      if (progress < 1) requestAnimationFrame(tick)
    }
    requestAnimationFrame(tick)
  }

  useEffect(() => {
    if (!ref.current) return

    // ① Si ya es visible al cargar → animar inmediatamente
    const rect = ref.current.getBoundingClientRect()
    if (rect.top < window.innerHeight) {
      animate(); return
    }

    // ② Si entra por scroll → animar al entrar
    const obs = new IntersectionObserver(
      ([entry]) => { if (entry.isIntersecting) { animate(); obs.disconnect() } },
      { threshold: 0.2 }
    )
    obs.observe(ref.current)

    // ③ Fallback: si pasan 2s sin animarse → mostrar valor final
    const timer = setTimeout(() => {
      if (!animated.current) { animate() }
    }, fallback)

    return () => { obs.disconnect(); clearTimeout(timer) }
  }, [ref])

  return value
}

src/components/StatsSection.tsx — usa el hook useCounter
'use client'
import { useRef } from 'react'
import { useCounter } from '@/hooks/useCounter'

export default function StatsSection() {
  const sectionRef = useRef<HTMLElement>(null)

  // Un ref por contador
  const r1 = useRef<HTMLDivElement>(null)
  const r2 = useRef<HTMLDivElement>(null)
  const r3 = useRef<HTMLDivElement>(null)
  const r4 = useRef<HTMLDivElement>(null)

  // Valores animados — target, duración, decimales
  const years     = useCounter(r1, { target: 4   })
  const followers = useCounter(r2, { target: 3, duration: 2000 })
  const posts     = useCounter(r3, { target: 166 })
  const rating    = useCounter(r4, { target: 4.8, decimals: 1 })

  return (
    <section className="stats-section" ref={sectionRef}>
      <div className="stats-grid">

        <div ref={r1} className="stat-item">
          <span className="stat-num">{years}</span>
          <p>Años endulzando Ciudad Real</p>
        </div>

        <div ref={r2} className="stat-item">
          <span className="stat-num">{followers}<span>K</span></span>
          <p>Comunidad en Instagram</p>
        </div>

        <div ref={r3} className="stat-item">
          <span className="stat-num">{posts}</span>
          <p>Publicaciones en redes</p>
        </div>

        <div ref={r4} className="stat-item">
          <span className="stat-num">{rating}<span>★</span></span>
          <p>Estrellas en Google Maps</p>
        </div>

      </div>
    </section>
  )
}
